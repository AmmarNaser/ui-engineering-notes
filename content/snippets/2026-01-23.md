This snippet demonstrates how to create a simple and reusable custom button component in React. The button can accept different styles and click events, making it versatile for various use cases.

### `CustomButton.js` - Reusable Button Component

```javascript
import React from 'react';
import PropTypes from 'prop-types';
import './CustomButton.css'; // Optional: You can add styles here

const CustomButton = ({ label, onClick, type = 'button', variant = 'primary' }) => {
    return (
        <button className={`custom-button ${variant}`} type={type} onClick={onClick}>
            {label}
        </button>
    );
};

CustomButton.propTypes = {
    label: PropTypes.string.isRequired,
    onClick: PropTypes.func.isRequired,
    type: PropTypes.oneOf(['button', 'submit', 'reset']),
    variant: PropTypes.oneOf(['primary', 'secondary', 'danger']),
};

export default CustomButton;
```

### Usage Example

```javascript
import React from 'react';
import ReactDOM from 'react-dom';
import CustomButton from './CustomButton';

const App = () => {
    const handleClick = () => {
        alert('Button clicked!');
    };

    return (
        <div>
            <h1>Custom Button Example</h1>
            <CustomButton label="Click Me!" onClick={handleClick} variant="primary" />
            <CustomButton label="Delete" onClick={handleClick} variant="danger" />
        </div>
    );
};

ReactDOM.render(<App />, document.getElementById('root'));
```
