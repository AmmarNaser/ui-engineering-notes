This code snippet demonstrates a reusable React component for a modal dialog that can be easily integrated into any application. The modal includes props for title, content, and visibility control.

### `Modal.js` - Reusable Modal Component

```jsx
import React from 'react';
import PropTypes from 'prop-types';
import './Modal.css'; // Assuming you've created a CSS file for styling

const Modal = ({ isOpen, title, content, onClose }) => {
  if (!isOpen) return null;

  return (
    <div className="modal-overlay">
      <div className="modal">
        <h2>{title}</h2>
        <p>{content}</p>
        <button onClick={onClose}>Close</button>
      </div>
    </div>
  );
};

Modal.propTypes = {
  isOpen: PropTypes.bool.isRequired,
  title: PropTypes.string.isRequired,
  content: PropTypes.string.isRequired,
  onClose: PropTypes.func.isRequired,
};

export default Modal;
```

### Usage Example

```jsx
import React, { useState } from 'react';
import Modal from './Modal';

const App = () => {
  const [isModalOpen, setModalOpen] = useState(false);

  const openModal = () => setModalOpen(true);
  const closeModal = () => setModalOpen(false);

  return (
    <div>
      <h1>Welcome to My App</h1>
      <button onClick={openModal}>Open Modal</button>
      <Modal 
        isOpen={isModalOpen} 
        title="My Modal Title" 
        content="This is the content of the modal." 
        onClose={closeModal} 
      />
    </div>
  );
};

export default App;
```
