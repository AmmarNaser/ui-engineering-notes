Here's a simple React code snippet for a custom hook that debounces a value. This can be useful for improving performance when handling user input, such as in search fields or form inputs.

```jsx
import { useState, useEffect } from 'react';

const useDebounce = (value, delay) => {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    // Cleanup function
    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
};

// Usage example in a functional component
const SearchInput = () => {
  const [inputValue, setInputValue] = useState('');
  const debouncedInputValue = useDebounce(inputValue, 500);

  const handleChange = (e) => {
    setInputValue(e.target.value);
  };

  useEffect(() => {
    if (debouncedInputValue) {
      // Trigger API call or any other side effect here
      console.log('Searching for:', debouncedInputValue);
    }
  }, [debouncedInputValue]);

  return (
    <input
      type="text"
      value={inputValue}
      onChange={handleChange}
      placeholder="Search..."
    />
  );
};

export default SearchInput;
```

### Explanation:
- The `useDebounce` hook takes a value and a delay as arguments. It updates the debounced value only after the specified delay when the value stops changing.
- The `SearchInput` component demonstrates how to use this hook by tracking user input and logging the debounced value, simulating a search functionality with a half-second delay.
