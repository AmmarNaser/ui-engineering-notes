This snippet demonstrates a reusable React component that implements a simple modal dialog. You can use this modal in any part of your application to display important information or gather user input.

### `Modal.js` - Reusable Modal Component

```javascript
import React, { useState } from 'react';
import './Modal.css'; // Assuming you have some basic styles here

const Modal = ({ isOpen, onClose, title, children }) => {
  if (!isOpen) return null;

  return (
    <div className="modal-overlay">
      <div className="modal">
        <h2>{title}</h2>
        <div className="modal-content">{children}</div>
        <button onClick={onClose}>Close</button>
      </div>
    </div>
  );
};

export default Modal;
```

### Usage Example

```javascript
import React, { useState } from 'react';
import Modal from './Modal';

const App = () => {
  const [isModalOpen, setModalOpen] = useState(false);

  const toggleModal = () => {
    setModalOpen(!isModalOpen);
  };

  return (
    <div>
      <h1>My App</h1>
      <button onClick={toggleModal}>Open Modal</button>
      <Modal
        isOpen={isModalOpen}
        onClose={toggleModal}
        title="My Modal"
      >
        <p>This is a reusable modal component!</p>
      </Modal>
    </div>
  );
};

export default App;
```
