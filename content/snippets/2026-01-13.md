### `useForm.js` - Custom Hook for Form Handling
```javascript
import { useState } from 'react';

/**
 * Custom hook for form input management
 * @param {Object} initialValues - Initial values for the form inputs
 * @returns {Array} - Array containing form values, handleChange function, and reset function
 */
const useForm = (initialValues) => {
  const [values, setValues] = useState(initialValues);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setValues({
      ...values,
      [name]: value,
    });
  };

  const reset = () => {
    setValues(initialValues);
  };

  return [values, handleChange, reset];
};

export default useForm;
```

### Usage Example
```javascript
import React from 'react';
import useForm from './useForm';

const MyForm = () => {
  const initialValues = { username: '', email: '' };
  const [formValues, handleChange, reset] = useForm(initialValues);

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('Form submitted:', formValues);
    reset(); // Reset form after submission
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>
          Username:
          <input
            type="text"
            name="username"
            value={formValues.username}
            onChange={handleChange}
          />
        </label>
      </div>
      <div>
        <label>
          Email:
          <input
            type="email"
            name="email"
            value={formValues.email}
            onChange={handleChange}
          />
        </label>
      </div>
      <button type="submit">Submit</button>
    </form>
  );
};

export default MyForm;
```

### Explanation
- **Custom Hook**: `useForm` provides a simple way to manage form state and handle input changes.
- **Initial Values**: Accepts an object with initial values for each form field.
- **Controlled Component**: Each input element is linked to the state managed by the hook, ensuring that they stay in sync.
- **Reset Functionality**: A reset function is included to clear the form back to initial values after submission.

This reusable hook simplifies form handling in React components, making it easier to manage input states and enhance user experience.
