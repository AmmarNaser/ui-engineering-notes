This snippet provides a customizable button component using React, allowing for different styles and click handlers to be passed as props.

### `CustomButton.js` - Reusable Custom Button Component
```jsx
import React from 'react';
import PropTypes from 'prop-types';
import './CustomButton.css'; // Optional: Custom styles can be added here

const CustomButton = ({ label, onClick, styleType }) => {
    return (
        <button className={`custom-button ${styleType}`} onClick={onClick}>
            {label}
        </button>
    );
};

CustomButton.propTypes = {
    label: PropTypes.string.isRequired,
    onClick: PropTypes.func.isRequired,
    styleType: PropTypes.string
};

CustomButton.defaultProps = {
    styleType: 'default' // Default style type
};

export default CustomButton;
```

### Usage Example
```jsx
import React from 'react';
import CustomButton from './CustomButton';

const App = () => {
    const handleClick = () => {
        alert('Button clicked!');
    };

    return (
        <div>
            <h1>My Custom Button</h1>
            <CustomButton label="Click Me!" onClick={handleClick} styleType="primary" />
        </div>
    );
};

export default App;
```
